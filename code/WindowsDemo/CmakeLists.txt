cmake_minimum_required(VERSION 3.10)

set(PROJECT_NAME WindowsDemo)

set(CMAKE_INSTALL_PREFIX ${CMAKE_CURRENT_SOURCE_DIR}/../../build/${PROJECT_NAME}/output CACHE PATH "Installation directory")

project(
    ${PROJECT_NAME}
    VERSION 1.0
    DESCRIPTION "A simple Hello World project"
    LANGUAGES C)

set(PROJECT_SOURCE "")
set(PROJECT_INCLUDE_DIRS "")

add_subdirectory(DemoSubFolder)

add_executable(${PROJECT_NAME} test.c)

target_link_libraries(${PROJECT_NAME} PRIVATE DemoSubFolder)

target_include_directories(
    ${PROJECT_NAME}
    PRIVATE
    ${PROJECT_INCLUDE_DIRS})

target_link_libraries(
    ${PROJECT_NAME}
    PRIVATE
    )

target_compile_options(
    ${PROJECT_NAME}
    PUBLIC
    ${CFLAGS})

target_link_options(
    ${PROJECT_NAME}
    PUBLIC
    ${LFLAGS})

message(STATUS "Operating System is: ${CMAKE_SYSTEM_NAME}")

install(TARGETS ${PROJECT_NAME} 
    RUNTIME DESTINATION bin
    LIBRARY DESTINATION lib
    ARCHIVE DESTINATION lib
    )
